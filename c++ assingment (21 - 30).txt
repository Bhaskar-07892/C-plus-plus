*(24)  WAP to demonstrate the use of References in c++.

#include <iostream>
#include <string>
using namespace std;

int main()
{
    string food = "Pizza";
    string &meal = food;
    
    cout << &meal << "\n";
    cout << food << "\n";
    cout << meal << "\n";
}





* WAP to demonstrate the use of deference in c++.

#include <iostream>
#include <string>
using namespace std;

int main()
{
    string food = "Pizza";
    string* ptr = &food;
    
    cout << ptr << "\n";
    cout << *ptr << "\n";
    cout << food << "\n";
}
