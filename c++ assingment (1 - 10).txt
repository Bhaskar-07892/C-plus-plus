* WAP to use all arithmetic Operators.

#include <iostream>
using namespace std;
int main()
{
     int x =20 ;
     int y = 5;
     
     // Addition
     cout  << x+y << "\n";
     
     // Substraction
     cout << x-y << "\n";
     
     // Multiplication
     cout << x*y << "\n";
     
     // Division
     cout << x/y << "\n";
     
     // Increment
     cout << ++x << "\n";
     
     // Decrement 
     cout << --y << "\n";
}




* WAP to use all relational Operators.

#include <iostream>
using namespace std;

int main()
{
    int x = 10, y = 15 , z = 10 , a = 15;
    
    //Equals to
    cout << (x == z) <<"\n";
    
    // Not Equals to
    cout << (x != y) << "\n";
    
    // Greater than
    cout << (x > z) << "\n";
    
    // Less than
    cout << (x < y) << "\n";
    
    // Greater than equal to
    cout << (x <= z) << "\n";
}




* WAP to use all logical Operators.

#include <iostream>
using namespace std;

int main()
{
    int x = 15;
    int y = 25;
    
    // AND Operator
    cout << (x > 10 && y < 30) << "\n";
    
    // OR Operator
    cout << (x <= 15 || y < 20) << "\n";
    
    // NOT Operator
    cout << !(x > 10 && y < 15) << "\n";
}
